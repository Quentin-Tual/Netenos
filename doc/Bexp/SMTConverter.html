def visit(node)
  case node
  when And
    visitAnd(node)
  when Or
    visitOr(node)
  when Not
    visitNot(node)
  when Operand
    visitOperand(node)
  else
    raise "Error: Unexpected class encountered #{node.class} during visit."
  end
end

def visitAnd(node)
  exp = ['(','and']
  node.operands.each do |op|
    exp += visit(op)
  end
  exp << ')'
end
