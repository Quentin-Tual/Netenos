def clean_working_dir circ_init, circ_test
    # * Move the VHD files, C&S script, VCD to a new directory named following the init circuit.
    Dir.mkdir(circ_init.name) unless File.exists?(circ_init.name)
    `mv #{circ_init.name}*.vhd #{circ_init.name}`
    `mv #{circ_init.name}*.vcd #{circ_init.name}`
    `mv compile_#{circ_init.name}.x #{circ_init.name}`
    `mv compile_#{circ_test.name}.x #{circ_init.name}`
    if @parameters[:dot_file]
        `mv #{circ_init.name}.dot #{circ_init.name}`
        `mv #{circ_test.name}.dot #{circ_init.name}`
    end
end

def gen_init_circ counter, delay_models
    # * Generate a netlist
    circ_init = @netlistGenerator.getRandomNetlist("rand_#{counter}")
    @layers_init = @netlistGenerator.grid
    # * Get data/caracteristics from the netlist
    @carac_init = @netlistGenerator.getNetlistInformations
    # * Generate the VHD files of the generated circuits
    @vhdl_converter.generate circ_init, delay_models[0]
    # * (optionnal) Generate a .dot file
    if @parameters[:dot_file]
        @viewer.dot circ_init
    end
    # * Recover adjacency matrix representing our netlist 
    @matrix_converter = Netlist::ConvNetlist2Matrix.new circ_init
    return circ_init
end

def gen_stim_seq method, circ_init, nb_sim_cycle
    case method
    when "sig_hammer"
        return Netlist::GenStim.new(circ_init).gen_sig_hammer_stim(nb_sim_cycle)#, trig_cond)
    else # "random" included
        return Netlist::GenStim.new(circ_init).gen_random_stim(nb_sim_cycle)#, trig_cond)
    end
end

def start 
    Dir.mkdir(@parameters[:path]) unless File.exists?(@parameters[:path])
    Dir.chdir(@parameters[:path]) do
        # * Generate GTECH  
        @vhdl_converter.gen_gtech

        # * For number of netlist to generate do :
        @parameters[:nb_netlist].times do |counter|
            
            @circ_init = gen_init_circ counter, @parameters[:delay_models]

            # * Initialization needed for later
            circ_test = nil
            @data[counter] = {}

            @inserter = Netlist::Tamperer.new(@circ_init.clone, @layers_init)

            # * Insert a certain amount of times and at differents places in circ_init the selected trojan 
            @parameters[:nb_insertion_per_netlist].times do |nth_insertion|
                
                @data[counter][nth_insertion] = {}
                @data[counter][nth_insertion][:ref_responses] = {}
                @data[counter][nth_insertion][:uuv_responses] = {}

                gen_test_circ
                stim_seq = gen_stim_seq @parameters[:stim_gen], @circ_init, @parameters[:nb_sim_cycle]
                gen_testbenchs @circ_init, @circ_test, @parameters[:test_frequencies].first, stim_seq
                gen_and_run_CS_script @circ_init, @circ_test, [@parameters[:test_frequencies].first]
                
                @traces_init, @traces_test = traces_recovery(
                    @circ_init,
                    @circ_test,
                    @parameters[:test_frequencies].first
                )

                # * Compare both traces
                # * Get the cycles index different between the two traces
                cycles_to_delete = @comparer.get_diff_cycle_num(@traces_init, @traces_test)
                # * Delete stimulis at those cycles to avoid HT Triggering
                stim_seq = @comparer.replace_cycle_list stim_seq, cycles_to_delete

                @parameters[:test_frequencies].each do |freq|
                    gen_testbenchs @circ_init, @circ_test, freq, stim_seq
                end

                gen_and_run_CS_script @circ_init, @circ_test, @parameters[:test_frequencies]

                # * Prepare data to store after the experiment (for ML use case)
                @data[counter][nth_insertion] = {:ref_adj_list => @matrix_converter.getAdjList}
                stim_seq.keys.each do |sig_name|
                    new_name = @matrix_converter.get_id_for_signame(sig_name)
                    stim_seq[new_name] = []
                    stim_seq[sig_name].each do |e|
                        stim_seq[new_name] << e.to_i
                    end
                    stim_seq.delete(sig_name)
                end
                @data[counter][nth_insertion][:stimulis] = stim_seq
                @data[counter][nth_insertion][:ref_responses] = {}
                @data[counter][nth_insertion][:uuv_responses] = {}

                @parameters[:test_frequencies].each do |freq|
                    @traces_init, @traces_test = traces_recovery(
                        @circ_init,
                        @circ_test,
                        freq
                    )

                    @results[freq] = @comparer.intercorrelation(@traces_test, @traces_init, @carac_init.last)

                    # * Prepare data for save (rename signals to be correspond with the matrix extracted : number IDs)
                    @traces_init.keys.each do |sig_name|
                        new_name = @matrix_converter.get_id_for_signame(sig_name.split("_")[1])
                        @traces_init[new_name] = []
                        @traces_init[sig_name].each do |e|
                            @traces_init[new_name] << e.to_i
                        end
                        @traces_init.delete(sig_name)
                    end
                    
                    @data[counter][nth_insertion][:ref_responses][freq] = @traces_init
                    @traces_test.keys.each do |sig_name|
                        new_name = @matrix_converter.get_id_for_signame(sig_name.split("_")[1])
                        @traces_test[new_name] = []
                        @traces_test[sig_name].each do |e|
                            @traces_test[new_name] << e.to_i
                        end
                        @traces_test.delete(sig_name)
                    end
                    @data[counter][nth_insertion][:uuv_responses][freq] = @traces_test
                end

                # * concat new results data to the rest in a results.csv file
                @parameters[:test_frequencies].each do |freq|
                    @result_csv << "#{@circ_test.name},#{freq.to_s.split('.').join},#{@results[freq]},#{@ht_layer},#{@carac_init[3]},#{cycles_to_delete.length}\n"
                end

            end # end insertion counter

            # * Save complete traces in a json file 
            File.write('data.json',Oj.dump(@data, :mode => :json))
            # * Save the results (detection performances for example) in a csv file 
            File.write('result.csv', @result_csv)

            clean_working_dir @circ_init, @circ_test

        end # end netlist counter
    end # end Dir env
end

def traces_recovery circ_init, circ_test, freq
    # * Recover simulation traces 
    @extractor_init.load_vcd "#{circ_init.name}_#{freq.to_s.split('.').join}_tb.vcd" # init circ traces PATH
    clk_period = @extractor_init.get_clock_period
    traces_init = @comparer.trace_to_list(@extractor_init.extract(@parameters[:compiler]), clk_period)
    @extractor_test.load_vcd  "#{circ_test.name}_#{freq.to_s.split('.').join}_tb.vcd" # test circ traces PATH
    clk_period_test = @extractor_test.get_clock_period
    if clk_period != clk_period_test 
        raise "Error : Different clock for the same targeted frequency."
    end
    traces_test = @comparer.trace_to_list(@extractor_test.extract(@parameters[:compiler]), clk_period)


    return traces_init, traces_test
end
