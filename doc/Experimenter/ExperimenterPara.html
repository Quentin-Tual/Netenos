def initialize parameters = nil
    if parameters.nil?
        # TODO : Set all parameters to default values 
        @parameters ={
            :path => "./exp_para",
            :dot_file => true,
            :test_frequencies => [1, 1.5, 2, 4, 8],
            # :test_frequencies => [1, 2, 4, 8],
            :nb_netlist => 10,
            :nb_insertion_per_netlist => 1,
            :nb_sim_cycle => 100,
            # :ht_type => "og_s38417",
            # :ht_type => "cotd_s38417",
            :ht_type => "xor_and",
            # :ht_type => "it_s38417",
            :nb_trig => 4,
            :netlist_carac => [20, 10, 10], # Mini ---> 36 gates
            # :netlist_carac => [50, 20, 10], # Tiny ---> 98 gates 
            # :netlist_carac => [100, 30, 20], # Small in the middle between both ---> 258 gates
            # :netlist_carac => [500, 150, 100], # Medium ---> ~2500 gates
            :delay_models => [:int,:int],
            :compiler => :ghdl,
            :jobs => 10,
            :verbose => true
        }
    else
        # * parse json file passed and set @parameters accordingly
        @parameters = Oj.load(File.read(parameters))
        @parameters.keys.each do |k|
            @parameters[k.to_sym] = @parameters[k]
            @parameters.delete k
        end
        @parameters[:compiler] = @parameters[:compiler].to_sym
        @parameters[:delay_models] = @parameters[:delay_models].collect {|e| e.to_sym}

    end # if 
    # * Verify any essential parameter is missing, else raise

    # * Initialize attributes necessary to run an experience
    @vhdl_converter = Converter::ConvNetlist2Vhdl.new
    @netlistGenerator = Netlist::RandomGenComb.new(
        @parameters[:netlist_carac][0], 
        @parameters[:netlist_carac][1], 
        @parameters[:netlist_carac][2]
    )
    @viewer = Converter::DotGen.new
    @matrix_converter = nil
    @inserter = nil
    @script_generator = Converter::VhdlCompiler.new
    @extractor_init = nil
    @extractor_test = nil
    @comparator = VCD::Vcd_Comparer.new
    @stim_seq = {}
    @id = nil

    @parameters[:nb_batch] = @parameters[:nb_netlist]/@parameters[:jobs] + (@parameters[:nb_netlist] % @parameters[:jobs] > 0 ? 1 : 0)
    @parameters[:batch_size] = @parameters[:nb_netlist]/@parameters[:nb_batch] + (@parameters[:nb_netlist] % @parameters[:nb_batch] > 0 ? 1 : 0)

    @circ_init = nil
    @layers_init = nil
    @carac_init = {}
    @traces_init = nil

    @circ_test = nil # TODO : insertions multiples -> passer en hash
    @ht_layer = {}
    @carac_test = nil
    @traces_test = nil
    

    @data = {}
    @results = {}
    @result_csv = "name,freq,correlation_based_similarity,ht_stage, nb_layer_init, expected,logic_testing_detections\n"

    if @parameters[:compiler] == :ghdl
        version = `ghdl -v`
        @parameters[:compiler] = @parameters[:compiler].to_s.concat(version[5]).to_sym
    end
end

def analyze 
    # * : Recover the vcd files from the simulation for each netlist pairs, compare it with intercorrelation, then save the results (result.csv) and the data from experience (data.json)
    
    @parameters[:nb_batch].times do |batch_id|
        Dir.chdir("batch#{batch_id}") do
            @parameters[:batch_size].times do |circ_id|
                Dir.chdir("circ#{circ_id}") do
                    @id = (batch_id.to_s + circ_id.to_s).to_i
                    @data[@id][:uuv_responses]={}
                    @data[@id][:ref_responses]={}
                    time = Time.new
                    puts "[+] Post-simulation steps for batch#{batch_id}/circ#{circ_id}     #{time.day}/#{time.month}/#{time.year} #{time.hour}:#{time.min}:#{time.
                    sec}"
                    @parameters[:test_frequencies].each do |freq|
                        puts " |--[+] Simulation traces recovery for frequency multiplier x#{freq}"
                        # TODO : utiliser 'traces_recovery' pour récupérer les traces
                        @traces_init, @traces_test = traces_recovery(batch_id,circ_id,freq)
                        puts " |--[+] Traces analysis for batch#{batch_id}/circ#{circ_id}"
                        # TODO : utiliser l'intercorrélation pour les comparer
                        @results[freq] = @comparator.intercorrelation(@traces_test,@traces_init, @carac_init[@id].last)

                        puts " |--[+] Data preparation for batch#{batch_id}/circ#{circ_id}"
                        # TODO : Préparer les données pour la sauvegarde
                        prepare_data 

                        puts " |--[+] Data saving for batch#{batch_id}/circ#{circ_id}"
                        # TODO : sauver les résultats et les données
                        save_data batch_id, circ_id, freq
                        save_results batch_id, circ_id, freq
                    end
                end
            end
        end

        time = Time.new
        puts "[+] Saving data        #{time.day}/#{time.month}/#{time.year} #{time.hour}:#{time.min}:#{time.sec}"
        # * Save complete traces in a json file 
        File.write('data.json',Oj.dump(@data, :mode => :json))
        # * Save the results (detection performances for example) in a csv file 
        File.write('result.csv', @result_csv)
    end

end

def build
    @script_generator.experience_makefile ".", @parameters[:nb_batch], @parameters[:jobs]
    gen_gtech_dir
    
    @parameters[:nb_batch].times do |batch_id|
        Dir.mkdir("batch#{batch_id}") unless File.exists?("batch#{batch_id}")
        Dir.chdir("batch#{batch_id}") do
            @script_generator.batch_makefile ".", @parameters[:batch_size]
            @parameters[:batch_size].times do |circ_id|
                Dir.mkdir("circ#{circ_id}") unless File.exists?("circ#{circ_id}")
                Dir.chdir("circ#{circ_id}") do
                    @id = (batch_id.to_s + circ_id.to_s).to_i
                    gen_circuits circ_id
                    gen_nominal_sim batch_id, circ_id
                end # chdir
            end # times
        end # chdir
    end # times

    # * Launch compilation and simulation at nominal frequency
    # @parameters[:verbose] ? `make` : `make -s`
    system "make -s"
    
    # * : Update stim sequence for each circuit
    @parameters[:nb_batch].times do |batch_id|
        @parameters[:batch_size].times do |circ_id|
            Dir.chdir("batch#{batch_id}/circ#{circ_id}") do
                @id = (batch_id.to_s + circ_id.to_s).to_i
                puts "[+] Complete simulation preparation batch#{batch_id}/circ#{circ_id}"
                update_stim batch_id, circ_id
                gen_all_sim batch_id, circ_id
            end
        end
    end

    # TODO : Sauver les stimulis dans data.json
end

def clean_working_dir circ_init, circ_test
    # * Move the VHD files, C&S script, VCD to a new directory named following the init circuit.
    Dir.mkdir(circ_init.name) unless File.exists?(circ_init.name)
    `mv #{circ_init.name}*.vhd #{circ_init.name}`
    `mv #{circ_init.name}*.vcd #{circ_init.name}`
    `mv compile_#{circ_init.name}.x #{circ_init.name}`
    `mv compile_#{circ_test.name}.x #{circ_init.name}`
    if @parameters[:dot_file]
        `mv #{circ_init.name}.dot #{circ_init.name}`
        `mv #{circ_test.name}.dot #{circ_init.name}`
    end # if
end

def gen_all_sim batch_id, circ_id
    # * : Generate testbench and compile scripts for simulation at all frequencies 
    puts " |--[+] Simulation generation at all frequencies for batch#{batch_id}/circ#{circ_id}"

    # * : Load the netlists :
    # TODO : Aussi possible de lister (genre de grep) tous les .enl et de les charger un à un, a voir pour palier au cas du copied plutôt que altered (2 cas diff à traiter rien de trop complexe en théorie) 
    file = IO.read("rand_#{circ_id}.enl")
    circ_test = Marshal.load(file)
    if File.exists?("rand_#{circ_id}_altered.enl") and File.exists?("rand_#{circ_id}_copied.enl")
        raise "Error : Found an altered and a copied version of the same netlist. It should not be possible. "
    elsif File.exists?("rand_#{circ_id}_altered.enl")
        file = IO.read("rand_#{circ_id}_altered.enl")
    else
        file = IO.read("rand_#{circ_id}_copied.enl")
    end
    circ_init = Marshal.load(file)

    # * : For each freq, generate associated testbench for circ_init and circ_test
    @parameters[:test_frequencies].each do |freq|
            print "     |-- "           
            gen_testbenchs circ_init, circ_test, freq, @stim_seq[@id]
    end # each

    # * Generate compile and simulate script
    gen_CS_scripts [circ_init.name, circ_test.name], @parameters[:test_frequencies], @parameters[:compiler]
end

def gen_init_circ circ_id, delay_models
    # * Generate a netlist
    circ_init = @netlistGenerator.getRandomNetlist("rand_#{circ_id}")
    # * Generate the VHD files of the generated circuits
    
    @vhdl_converter.generate circ_init, delay_models[0]
    @layers_init = @netlistGenerator.grid
    # * Get data/caracteristics from the netlist
    @carac_init[@id] = @netlistGenerator.getNetlistInformations
    # puts "Gates amount : #{@carac_init[@id][2]}"
    
    # * (optionnal) Generate a .dot file
    if @parameters[:dot_file]
        @viewer.dot circ_init
    end # if
    # * Recover adjacency matrix representing our netlist 
    @matrix_converter = Converter::ConvNetlist2Matrix.new circ_init

    circ_init.save_as("./")

    return circ_init
end

    # * Prepare data for save (rename signals to be correspond with the matrix extracted : number IDs)
    @traces_init.keys.each do |sig_name|

end

def start
    start_time = Time.new
    puts "[+] Starting experience : #{start_time.day}/#{start_time.month}/#{start_time.year} #{start_time.hour}:#{start_time.min}:#{start_time.
    sec}"
    if File.exists?(@parameters[:path])
        puts "[-] Warning : Experience directory \"#{@parameters[:path]}\" already existing.\n           Press Enter to confirm deletion...\n"
        gets
        FileUtils.rm_rf(@parameters[:path])
    end
    Dir.mkdir(@parameters[:path])
    Dir.chdir(@parameters[:path]) do 
        export_config "./e1-tiny-xor_and.conf"
        time = Time.new
        puts "[+] Starting build step : #{time.day}/#{time.month}/#{time.year} #{time.hour}:#{time.min}:#{time.
    sec}"
        build
        time = Time.new
        puts "[+] Launching makefile for all frequencies simulation : #{time.day}/#{time.month}/#{time.year} #{time.hour}:#{time.min}:#{time.
    sec}"
        system "make -s"
        time = Time.new
        puts "[+] Starting analysis step : #{time.day}/#{time.month}/#{time.year} #{time.hour}:#{time.min}:#{time.
        sec}"
        analyze

        time = Time.new
        puts "+-------------------------------------------------------"
        puts "| Experience started at #{start_time.day}/#{start_time.month}/#{start_time.year} #{start_time.hour}:#{start_time.min}:#{start_time.
        sec}"
        puts "| Finished at #{time.day}/#{time.month}/#{time.year} #{time.hour}:#{time.min}:#{time.sec}"
        puts "+-------------------------------------------------------"
    end
end

def traces_recovery batch_id, circ_id, freq
    # * Recover simulation traces 
    @extractor_init = VCD::Vcd_Signal_Extractor.new
    @extractor_init.load_vcd "rand_#{circ_id}_#{freq.to_s.split('.').join}_tb.vcd" # init circ traces PATH

    if File.exists?("rand_#{circ_id}_altered.vhd") 
        test_type = "altered"
    else 
        test_type = "copied"
    end
    @extractor_test = VCD::Vcd_Signal_Extractor.new
    @extractor_test.load_vcd  "rand_#{circ_id}_#{test_type}_#{freq.to_s.split('.').join}_tb.vcd" 
    # !!! "altered" or "copied" workaround to find

    clk_period = @extractor_init.get_clock_period
    traces_init = @comparator.trace_to_list(@extractor_init.extract(@parameters[:compiler]), clk_period)           

    # test circ traces PATH
    clk_period_test = @extractor_test.get_clock_period
    if clk_period != clk_period_test 
        raise "Error : Different clock for the same targeted frequency."
    end # if
    traces_test = @comparator.trace_to_list(@extractor_test.extract(@parameters[:compiler]), clk_period)

    return traces_init, traces_test
end

def update_stim batch_id, circ_id
    puts " |--[+] Stimuli sequence update for batch#{batch_id}/circ#{circ_id}"
    # * : Recover and analyze nominal simulation traces then replace triggering stimulis from stim seq
    @traces_init, @traces_test = traces_recovery(batch_id, circ_id, @parameters[:test_frequencies].first)

    # * Compare both traces
    # * Get the cycles index different between the two traces
    cycles_to_delete = @comparator.get_diff_cycle_num(@traces_init, @traces_test)
    # * Delete stimulis at those cycles to avoid HT Triggering
    @stim_seq[@id] = @comparator.replace_cycle_list @stim_seq[@id], cycles_to_delete
    @data[@id] = {:stimulis => @stim_seq[@id]} # * : store stim sequence for each id in data  
end
