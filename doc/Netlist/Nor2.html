def compute_transit_proba transi_proba, rounding = 5
    proba_i0, proba_i1 = transi_proba
    ((1.0 - (1.0 - proba_i0) * (1.0 - proba_i1)) * (1.0 - proba_i0) * (1.0 - proba_i1)).round(rounding)
end
