def afterward_value
    case value
    when :R, :S1
        :S1
    when :F, :S0
        :S0
    else
        raise "Error : Unknown transition value encountered. Cannot obtain afterward value of #{value}."
    end
end

def closest_inferior_timestamp events
    events.select{|e| e.timestamp <= timestamp}.min_by{|e| (e.timestamp - timestamp).abs}
end
