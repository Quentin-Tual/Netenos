def get_input_transition output_transition
  case output_transition
  when "0" 
      return [["0","0"]]
  when "1" 
      return [["0","1"],["1","0"],["1","1"],["1","F"],["F","1"],["1","R"],["R","1"],["R","F"],["F","R"]]
  when "R"
      return [["0","R"],["R","0"],["R","R"]]
  when "F"
      return [["0","F"],["F","0"],["F","F"]]
  else
      raise "Error: Unexpected output transition value encountered."
  end
end
