def apply_fun fun
  values = @subnodes.collect do |delayNode|
    delayNode.attr_flat_float_list
  end.flatten
  if fun == :avg or fun == :mean
    (values.sum / values.size).round(3)
  else # :min or :max
    values.send(fun)
  end
end
