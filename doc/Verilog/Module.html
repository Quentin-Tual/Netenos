def add node 
  case node
  when Wire
    @wires << node
  when Input
    @inputs << node
  when Output
    @outputs << node
  when Instance 
    @instances << node
  else 
    raise "Error: Unknown type #{node.class} to add in a #{self.class}"
  end
end
